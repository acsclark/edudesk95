  <script>
    // --- Global Game State ---
    let words = [];
    let level = 1;
    let currentWordIndex = 0;
    let collectedCount = 0;
    
    // Engine State
    const state = {
      mouse: { x: 100, y: 100, radius: 18, speed: 4, dx: 0, dy: 0, angle: 0 },
      cat: { x: 300, y: 200, radius: 24, speed: 1.5, angle: 0, active: false },
      house: { x: 700, y: 500, radius: 40 },
      cheeses: [],
      particles: [],
      keys: { up: false, down: false, left: false, right: false },
      lastTime: 0,
      cameraShake: 0,
      levelComplete: false,
      isDead: false,
      animationFrameId: null
    };

    // --- DOM Elements ---
    const screens = {
      menu: document.getElementById('screen-menu'),
      interlude: document.getElementById('screen-interlude'),
      finished: document.getElementById('screen-finished'),
      game: document.getElementById('screen-game')
    };

    const canvas = document.getElementById('game-canvas');
    const ctx = canvas.getContext('2d');
    const hudLetters = document.getElementById('hud-letters');
    const hudLevelText = document.getElementById('hud-level-text');
    const hudStatus = document.getElementById('hud-status');
    const feedbackText = document.getElementById('feedback-text');

    // --- Navigation Functions ---
    function showScreen(screenName) {
      Object.values(screens).forEach(s => s.classList.add('hidden'));
      screens[screenName].classList.remove('hidden');
    }

    function setFeedback(msg) {
      feedbackText.innerText = msg;
    }

    function updateHUD() {
      const currentWord = words[currentWordIndex] || '';
      const isReturningHome = collectedCount >= currentWord.length;

      hudLevelText.innerText = `Level ${level} - Word ${currentWordIndex + 1} of ${words.length}`;
      
      if (isReturningHome) {
        hudStatus.innerText = 'Go Home!';
        hudStatus.className = 'text-sm font-bold text-emerald-500 animate-bounce';
      } else {
        hudStatus.innerText = 'Hunting';
        hudStatus.className = 'text-sm font-bold text-blue-500';
      }

      hudLetters.innerHTML = '';
      for (let i = 0; i < currentWord.length; i++) {
        const div = document.createElement('div');
        div.className = `w-10 h-12 flex items-center justify-center text-2xl font-black rounded-lg border-b-4 `;
        
        if (i < collectedCount) {
          div.className += 'bg-amber-400 text-white border-amber-500';
          div.innerText = currentWord[i];
        } else if (i === collectedCount && !isReturningHome) {
          div.className += 'bg-blue-100 border-blue-300 text-blue-400 animate-pulse';
          div.innerText = '_';
        } else {
          div.className += 'bg-slate-100 border-slate-200 text-slate-300';
          div.innerText = '_';
        }
        hudLetters.appendChild(div);
      }
    }

    // --- Game Engine Logic ---
    function initLevel(word) {
      state.levelComplete = false;
      state.isDead = false;
      state.mouse.x = 60;
      state.mouse.y = 60;
      state.mouse.dx = 0;
      state.mouse.dy = 0;
      state.house.x = canvas.width - 60;
      state.house.y = canvas.height - 60;

      state.cat.x = canvas.width / 2;
      state.cat.y = canvas.height / 2;
      state.cat.active = (level >= 2);

      const newCheeses = [];
      for (let i = 0; i < word.length; i++) {
        let placed = false;
        let attempts = 0;
        while (!placed && attempts < 100) {
          attempts++;
          const cx = Math.random() * (canvas.width - 120) + 60;
          const cy = Math.random() * (canvas.height - 120) + 60;
          
          const distMouse = Math.hypot(cx - 60, cy - 60);
          const distHouse = Math.hypot(cx - (canvas.width - 60), cy - (canvas.height - 60));
          
          if (distMouse > 120 && distHouse > 120) {
            const overlap = newCheeses.some(c => Math.hypot(cx - c.x, cy - c.y) < 60);
            if (!overlap) {
              newCheeses.push({ x: cx, y: cy, radius: 22, letter: word[i], orderIndex: i, active: true, scale: 0 });
              placed = true;
            }
          }
        }
      }
      state.cheeses = newCheeses;
      collectedCount = 0;
      setFeedback('Find the first letter!');
      updateHUD();
    }

    function startGame() {
      if (state.animationFrameId) cancelAnimationFrame(state.animationFrameId);
      currentWordIndex = 0;
      initLevel(words[currentWordIndex]);
      state.lastTime = performance.now();
      gameLoop(state.lastTime);
    }

    function handleWordComplete() {
      setFeedback("Word Complete!");
      setTimeout(() => {
        if (currentWordIndex + 1 >= words.length) {
          // Finished all words for this level
          cancelAnimationFrame(state.animationFrameId);
          if (level === 1) {
            showScreen('interlude');
          } else {
            showScreen('finished');
          }
        } else {
          currentWordIndex++;
          initLevel(words[currentWordIndex]);
        }
      }, 800);
    }

    function update(deltaTime) {
      if (state.isDead) return;
      const currentWord = words[currentWordIndex] || '';
      const { mouse, house, cheeses, keys } = state;

      // Input
      if (keys.up) mouse.dy -= 0.5;
      if (keys.down) mouse.dy += 0.5;
      if (keys.left) mouse.dx -= 0.5;
      if (keys.right) mouse.dx += 0.5;

      mouse.dx *= 0.85;
      mouse.dy *= 0.85;

      const speed = Math.hypot(mouse.dx, mouse.dy);
      if (speed > mouse.speed) {
        mouse.dx = (mouse.dx / speed) * mouse.speed;
        mouse.dy = (mouse.dy / speed) * mouse.speed;
      }

      mouse.x += mouse.dx;
      mouse.y += mouse.dy;
      if (speed > 0.5) mouse.angle = Math.atan2(mouse.dy, mouse.dx);

      const margin = mouse.radius;
      if (mouse.x < margin) { mouse.x = margin; mouse.dx *= -0.5; }
      if (mouse.x > canvas.width - margin) { mouse.x = canvas.width - margin; mouse.dx *= -0.5; }
      if (mouse.y < margin) { mouse.y = margin; mouse.dy *= -0.5; }
      if (mouse.y > canvas.height - margin) { mouse.y = canvas.height - margin; mouse.dy *= -0.5; }

      if (state.cameraShake > 0) {
        state.cameraShake -= deltaTime * 0.05;
        if (state.cameraShake < 0) state.cameraShake = 0;
      }

      // Cat Logic
      if (state.cat.active && !state.levelComplete) {
        const dx = mouse.x - state.cat.x;
        const dy = mouse.y - state.cat.y;
        const dist = Math.hypot(dx, dy);
        if (dist > 0) {
          state.cat.x += (dx / dist) * state.cat.speed;
          state.cat.y += (dy / dist) * state.cat.speed;
          state.cat.angle = Math.atan2(dy, dx);
        }

        if (dist < mouse.radius + state.cat.radius) {
          state.isDead = true;
          state.cameraShake = 10;
          setFeedback("The cat caught you! Try again.");
          for(let i=0; i<20; i++) {
            state.particles.push({
              x: mouse.x, y: mouse.y,
              vx: (Math.random() - 0.5) * 15, vy: (Math.random() - 0.5) * 15,
              life: 1, size: Math.random() * 6 + 2, color: '#ef4444'
            });
          }
          setTimeout(() => initLevel(currentWord), 1500);
          return;
        }
      }

      cheeses.forEach(c => {
        if (c.active && c.scale < 1) c.scale = Math.min(1, c.scale + 0.05);
      });

      state.particles = state.particles.filter(p => p.life > 0);
      state.particles.forEach(p => {
        p.x += p.vx;
        p.y += p.vy;
        p.life -= deltaTime * 0.01;
        p.size *= 0.95;
      });

      const currentTargetIndex = cheeses.filter(c => !c.active).length;
      
      cheeses.forEach(c => {
        if (c.active) {
          const dist = Math.hypot(mouse.x - c.x, mouse.y - c.y);
          if (dist < mouse.radius + c.radius) {
            if (c.orderIndex === currentTargetIndex) {
              c.active = false;
              collectedCount = currentTargetIndex + 1;
              updateHUD();
              
              if (collectedCount >= currentWord.length) setFeedback("Great! Now go to the house!");
              else setFeedback("Got it! Find the next one.");
              
              for(let i=0; i<10; i++) {
                state.particles.push({
                  x: c.x, y: c.y,
                  vx: (Math.random() - 0.5) * 10, vy: (Math.random() - 0.5) * 10,
                  life: 1, size: Math.random() * 5 + 3, color: '#FBBF24'
                });
              }
            } else {
              mouse.dx *= -2;
              mouse.dy *= -2;
              state.cameraShake = 5;
              setFeedback("Oops! Wrong letter!");
            }
          }
        }
      });

      if (currentTargetIndex >= currentWord.length && !state.levelComplete) {
        const dist = Math.hypot(mouse.x - house.x, mouse.y - house.y);
        if (dist < mouse.radius + house.radius) {
          state.levelComplete = true;
          handleWordComplete();
        }
      }
    }

    function draw() {
      const currentWord = words[currentWordIndex] || '';
      const { mouse, house, cheeses, particles, cameraShake } = state;

      ctx.save();
      if (cameraShake > 0) {
        const dx = (Math.random() - 0.5) * cameraShake;
        const dy = (Math.random() - 0.5) * cameraShake;
        ctx.translate(dx, dy);
      }

      // Background
      ctx.fillStyle = '#a7f3d0';
      ctx.fillRect(0, 0, canvas.width, canvas.height);

      // Grid
      ctx.strokeStyle = '#6ee7b7';
      ctx.lineWidth = 2;
      for(let i=0; i<canvas.width; i+=40) { ctx.beginPath(); ctx.moveTo(i, 0); ctx.lineTo(i, canvas.height); ctx.stroke(); }
      for(let i=0; i<canvas.height; i+=40) { ctx.beginPath(); ctx.moveTo(0, i); ctx.lineTo(canvas.width, i); ctx.stroke(); }

      // House
      const currentTargetIndex = cheeses.filter(c => !c.active).length;
      const homeActive = currentTargetIndex >= currentWord.length;
      
      ctx.save();
      ctx.translate(house.x, house.y);
      if (homeActive) {
        ctx.beginPath();
        ctx.arc(0, 0, house.radius + 10 + Math.sin(Date.now() / 150) * 5, 0, Math.PI * 2);
        ctx.fillStyle = 'rgba(252, 211, 77, 0.5)';
        ctx.fill();
      }
      ctx.fillStyle = '#78350f';
      ctx.fillRect(-house.radius - 20, -house.radius, (house.radius + 20) * 2, house.radius * 2);
      ctx.fillStyle = '#b45309';
      ctx.fillRect(-house.radius - 20, -house.radius, (house.radius + 20) * 2, 10);
      ctx.beginPath();
      ctx.arc(0, house.radius, house.radius, Math.PI, 0);
      ctx.fillStyle = '#000000';
      ctx.fill();
      ctx.restore();

      // Cheeses
      cheeses.forEach(c => {
        if (c.active) {
          ctx.save();
          ctx.translate(c.x, c.y);
          ctx.scale(c.scale, c.scale);
          ctx.translate(0, Math.sin(Date.now() / 200 + c.orderIndex) * 3);

          ctx.beginPath();
          ctx.moveTo(-c.radius, c.radius); ctx.lineTo(c.radius, c.radius); ctx.lineTo(0, -c.radius);
          ctx.closePath();
          ctx.fillStyle = '#fbbf24'; ctx.fill();
          ctx.lineWidth = 3; ctx.strokeStyle = '#d97706'; ctx.stroke();

          ctx.fillStyle = '#d97706';
          ctx.beginPath(); ctx.arc(-5, 5, 3, 0, Math.PI*2); ctx.fill();
          ctx.beginPath(); ctx.arc(8, 8, 4, 0, Math.PI*2); ctx.fill();
          ctx.beginPath(); ctx.arc(0, -5, 2, 0, Math.PI*2); ctx.fill();

          ctx.fillStyle = '#1e293b';
          ctx.font = 'bold 24px sans-serif';
          ctx.textAlign = 'center'; ctx.textBaseline = 'middle';
          if (c.orderIndex === currentTargetIndex) {
            ctx.fillStyle = '#dc2626'; ctx.font = 'bold 28px sans-serif';
            ctx.shadowColor = "white"; ctx.shadowBlur = 10;
          }
          ctx.fillText(c.letter, 0, 5);
          ctx.shadowBlur = 0;
          ctx.restore();
        }
      });

      // Particles
      particles.forEach(p => {
        ctx.beginPath(); ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
        ctx.fillStyle = p.color; ctx.globalAlpha = p.life; ctx.fill(); ctx.globalAlpha = 1.0;
      });

      // Cat
      if (state.cat.active) {
        ctx.save();
        ctx.translate(state.cat.x, state.cat.y);
        ctx.rotate(state.cat.angle);

        ctx.beginPath(); ctx.moveTo(-state.cat.radius, 0);
        const catTailWiggle = Math.sin(Date.now() / 150) * 15;
        ctx.quadraticCurveTo(-state.cat.radius - 30, catTailWiggle, -state.cat.radius - 50, catTailWiggle);
        ctx.strokeStyle = '#ea580c'; ctx.lineWidth = 6; ctx.stroke();

        ctx.beginPath(); ctx.ellipse(0, 0, state.cat.radius, state.cat.radius * 0.75, 0, 0, Math.PI * 2);
        ctx.fillStyle = '#f97316'; ctx.fill(); ctx.strokeStyle = '#c2410c'; ctx.lineWidth = 2; ctx.stroke();

        ctx.beginPath(); ctx.moveTo(10, -10); ctx.lineTo(0, -25); ctx.lineTo(-10, -10); ctx.fillStyle = '#f97316'; ctx.fill(); ctx.stroke();
        ctx.beginPath(); ctx.moveTo(10, 10); ctx.lineTo(0, 25); ctx.lineTo(-10, 10); ctx.fillStyle = '#f97316'; ctx.fill(); ctx.stroke();

        ctx.beginPath(); ctx.moveTo(8, -12); ctx.lineTo(2, -20); ctx.lineTo(-4, -12); ctx.fillStyle = '#fca5a5'; ctx.fill();
        ctx.beginPath(); ctx.moveTo(8, 12); ctx.lineTo(2, 20); ctx.lineTo(-4, 12); ctx.fillStyle = '#fca5a5'; ctx.fill();

        ctx.fillStyle = '#fef08a';
        ctx.beginPath(); ctx.arc(12, -8, 4, 0, Math.PI * 2); ctx.fill();
        ctx.beginPath(); ctx.arc(12, 8, 4, 0, Math.PI * 2); ctx.fill();
        ctx.fillStyle = '#000';
        ctx.beginPath(); ctx.ellipse(13, -8, 1, 3, 0, 0, Math.PI*2); ctx.fill();
        ctx.beginPath(); ctx.ellipse(13, 8, 1, 3, 0, 0, Math.PI*2); ctx.fill();
        
        ctx.fillStyle = '#ef4444';
        ctx.beginPath(); ctx.arc(state.cat.radius, 0, 3, 0, Math.PI * 2); ctx.fill();
        ctx.restore();
      }

      // Mouse
      if (!state.isDead) {
        ctx.save();
        ctx.translate(mouse.x, mouse.y);
        ctx.rotate(mouse.angle);

        ctx.beginPath(); ctx.moveTo(-mouse.radius, 0);
        const tailWiggle = Math.sin(Date.now() / 100) * 10 * (Math.hypot(mouse.dx, mouse.dy) > 0.5 ? 1 : 0.2);
        ctx.quadraticCurveTo(-mouse.radius - 20, tailWiggle, -mouse.radius - 40, tailWiggle/2);
        ctx.strokeStyle = '#f472b6'; ctx.lineWidth = 4; ctx.stroke();

        ctx.beginPath(); ctx.ellipse(0, 0, mouse.radius, mouse.radius * 0.7, 0, 0, Math.PI * 2);
        ctx.fillStyle = '#94a3b8'; ctx.fill(); ctx.strokeStyle = '#475569'; ctx.lineWidth = 2; ctx.stroke();

        ctx.beginPath(); ctx.arc(5, -12, 8, 0, Math.PI*2); ctx.fillStyle = '#cbd5e1'; ctx.fill(); ctx.stroke();
        ctx.beginPath(); ctx.arc(5, -12, 4, 0, Math.PI*2); ctx.fillStyle = '#f472b6'; ctx.fill();
        ctx.beginPath(); ctx.arc(5, 12, 8, 0, Math.PI*2); ctx.fillStyle = '#cbd5e1'; ctx.fill(); ctx.stroke();
        ctx.beginPath(); ctx.arc(5, 12, 4, 0, Math.PI*2); ctx.fillStyle = '#f472b6'; ctx.fill();

        ctx.fillStyle = '#0f172a';
        ctx.beginPath(); ctx.arc(10, -5, 2, 0, Math.PI*2); ctx.fill();
        ctx.beginPath(); ctx.arc(10, 5, 2, 0, Math.PI*2); ctx.fill();

        ctx.fillStyle = '#ec4899';
        ctx.beginPath(); ctx.arc(mouse.radius, 0, 3, 0, Math.PI*2); ctx.fill();

        ctx.strokeStyle = '#e2e8f0'; ctx.lineWidth = 1;
        ctx.beginPath(); ctx.moveTo(mouse.radius, 2); ctx.lineTo(mouse.radius + 15, 10); ctx.stroke();
        ctx.beginPath(); ctx.moveTo(mouse.radius, -2); ctx.lineTo(mouse.radius + 15, -10); ctx.stroke();
        ctx.restore();
      }
      ctx.restore();
    }

    function gameLoop(time) {
      const deltaTime = time - state.lastTime;
      state.lastTime = time;
      update(deltaTime);
      draw();
      state.animationFrameId = requestAnimationFrame(gameLoop);
    }

    // --- Input Handling ---
    window.addEventListener('keydown', e => {
      // Prevent default browser scrolling for arrow keys and spacebar
      if (['ArrowUp', 'ArrowDown', 'ArrowLeft', 'ArrowRight', ' '].includes(e.key)) {
        e.preventDefault();
      }
      
      switch(e.key) {
        case 'ArrowUp': case 'w': state.keys.up = true; break;
        case 'ArrowDown': case 's': state.keys.down = true; break;
        case 'ArrowLeft': case 'a': state.keys.left = true; break;
        case 'ArrowRight': case 'd': state.keys.right = true; break;
      }
    });

    window.addEventListener('keyup', e => {
      // Prevent default on keyup too just in case
      if (['ArrowUp', 'ArrowDown', 'ArrowLeft', 'ArrowRight', ' '].includes(e.key)) {
        e.preventDefault();
      }

      switch(e.key) {
        case 'ArrowUp': case 'w': state.keys.up = false; break;
        case 'ArrowDown': case 's': state.keys.down = false; break;
        case 'ArrowLeft': case 'a': state.keys.left = false; break;
        case 'ArrowRight': case 'd': state.keys.right = false; break;
      }
    });

    document.querySelectorAll('.dpad-btn').forEach(btn => {
      const dir = btn.getAttribute('data-dir');
      btn.addEventListener('pointerdown', e => { e.preventDefault(); state.keys[dir] = true; });
      btn.addEventListener('pointerup', e => { e.preventDefault(); state.keys[dir] = false; });
      btn.addEventListener('pointercancel', e => { e.preventDefault(); state.keys[dir] = false; });
    });

    // --- Button Actions ---
    document.getElementById('btn-start').addEventListener('click', () => {
      const inputText = document.getElementById('word-input').value;
      const parsedWords = inputText.split('\n')
        .map(w => w.trim().toUpperCase())
        .filter(w => w.length > 0 && /^[A-Z]+$/.test(w));
      
      if (parsedWords.length > 0) {
        words = parsedWords;
        level = 1;
        showScreen('game');
        startGame();
      } else {
        alert('Please enter at least one valid word (letters only).');
      }
    });

    document.getElementById('btn-level2').addEventListener('click', () => {
      level = 2;
      showScreen('game');
      startGame();
    });

    document.getElementById('btn-restart').addEventListener('click', () => {
      level = 1;
      showScreen('menu');
    });

    document.getElementById('btn-quit').addEventListener('click', () => {
      if (state.animationFrameId) cancelAnimationFrame(state.animationFrameId);
      showScreen('menu');
    });
